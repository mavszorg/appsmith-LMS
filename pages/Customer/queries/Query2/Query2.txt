SELECT c1."JoiningDate", c1."OrgID", c1."CustomerName", c1."Country", c1."IsKYCDone", c1."RejectList", c1."WithIMB", c1."Id", c2."PlanDealID", c2."ProductID", c2."Rate", c2."Date", c2."CustomerID", p."Plan", p."Limit", p1."ProductDescription", p1."CreatedAt", p1."MailType", p1."ProductType", p1."QtyInStock"
FROM "public"."Customers" c1 
	INNER JOIN "public"."CustomerPrice" c2 ON ( c2."CustomerID" = c1."Id"  )  
	LEFT OUTER JOIN "public"."PlanDeal" p ON ( p."Id" = c2."PlanDealID"  )  
	INNER JOIN "public"."Products" p1 ON ( p1."Id" = c2."ProductID"  )  
WHERE  c2."Rate" IS NULL
